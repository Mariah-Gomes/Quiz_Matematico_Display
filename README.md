void playNote(int noteFrequency, int noteDuration) {

    delay(noteDuration);

void setup() {

void loop() {
  // Iniciar jogo.
  if(iniciar == 1){
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Game ON!");
    delay(2000);
    lcd.clear();

    }else if(contAcertosGeral >= 5){

      	verificacaoNoob();

      	verificacaoBoa();

          verificacaoNoob();
      	}else if(contAcertosGeral >= 15){
          verificacaoBoa();
          // Chamando funcao da pergunta final.

      }

    iniciar = 0;
  	nivel = 0;

    contAcertosGeral = 0;

    score = 0;
    // Resetando array "jaFoi" caso comecem outro jogo.
    for(int i = 0; i < 24; i++){
      jaFoi[i] = -1;
    }
  }
}
